# Let's use the handy function we created
def adj_r2(x1,y):
    r2 = reg.score(x1,y)
    n = x1.shape[0]
    p = x1.shape[1]
    adjusted_r2 = 1-(1-r2)*(n-1)/(n-p-1)
    return adjusted_r2
